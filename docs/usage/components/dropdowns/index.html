<?php


// Create form
$factory = new \Laminas\Form\Factory();
$form    = $factory->create(
    [
        'type'       => 'form',
        'name'       => 'dropdown',
        'attributes' => ['id' => 'dropdown'],
        'elements'   => [
            [
                'spec' => [
                    'name'       => 'email',
                    'options'    => ['label' => 'Email address'],
                    'attributes' => [
                        'type'        => 'email',
                        'id'          => 'exampleDropdownFormEmail1',
                        'placeholder' => 'email@example.com',
                    ],
                ],
            ],
            [
                'spec' => [
                    'name'       => 'password',
                    'options'    => ['label' => 'Password'],
                    'attributes' => [
                        'type'        => 'password',
                        'id'          => 'exampleDropdownFormPassword1',
                        'placeholder' => 'Password',
                    ],
                ],
            ],
            [
                'spec' => [
                    'type'       => 'checkbox',
                    'name'       => 'remember_me',
                    'options'    => [
                        'label'              => 'Remember me',
                        'use_hidden_element' => false,
                    ],
                    'attributes' => [    'id' => 'dropdownCheck'    ],
                ],
            ],
            [
                'spec' => [
                    'type'    => 'submit',
                    'options' => [
                        'label'   => 'Sign in',
                        'variant' => 'primary',
                    ],
                ],
            ],
        ],
    ]
);

echo $this->dropdown()->renderMenu(
    [
        $form,
        'New around here? Sign up',
        'Forgot password?',
    ]
);
